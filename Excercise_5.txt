DECLARE
    FIRST NUMBER:=0;
    SECOND NUMBER:=1;
    TEMP NUMBER;
    N NUMBER:=8;
    I NUMBER;
BEGIN
    DBMS_OUTPUT.PUT_LINE('FIBONACCI SERIES:');
    DBMS_OUTPUT.PUT_LINE(FIRST);
    DBMS_OUTPUT.PUT_LINE(SECOND);
    FOR I IN 2..N-1 LOOP
        TEMP:=FIRST+SECOND;
        FIRST:=SECOND;
        SECOND:=TEMP;
        DBMS_OUTPUT.PUT_LINE(TEMP);
    END LOOP;
END;
